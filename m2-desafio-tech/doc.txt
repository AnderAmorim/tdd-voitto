Desafio 2 - Primeiros testes unitários no React e Simplificando testes com Testing Library:

Primeiro, instale as dependências necessárias:

Copy code
npm install react react-dom react-testing-library jest
Em seguida, crie um arquivo de teste chamado contador.test.js e adicione o seguinte código:

javascript
Copy code
import React from 'react';
import { render, fireEvent } from '@testing-library/react';
import Contador from './Contador';

test('renderiza o componente Contador corretamente', () => {
  const { getByText } = render(<Contador />);
  const valor = getByText(/Valor:/);
  const botaoIncrementar = getByText('+');
  const botaoDecrementar = getByText('-');

  expect(valor).toBeInTheDocument();
  expect(botaoIncrementar).toBeInTheDocument();
  expect(botaoDecrementar).toBeInTheDocument();
});

test('incrementa o valor corretamente ao clicar no botão de incremento', () => {
  const { getByText } = render(<Contador />);
  const valor = getByText(/Valor:/);
  const botaoIncrementar = getByText('+');

  fireEvent.click(botaoIncrementar);

  expect(valor.textContent).toBe('Valor: 1');
});

test('decrementa o valor corretamente ao clicar no botão de decremento', () => {
  const { getByText } = render(<Contador />);
  const valor = getByText(/Valor:/);
  const botaoDecrementar = getByText('-');

  fireEvent.click(botaoDecrementar);

  expect(valor.textContent).toBe('Valor: -1');
});

test('o valor nunca fica negativo', () => {
  const { getByText } = render(<Contador />);
  const valor = getByText(/Valor:/);
  const botaoDecrementar = getByText('-');

  fireEvent.click(botaoDecrementar);
  fireEvent.click(botaoDecrementar);

  expect(valor.textContent).toBe('Valor: 0');
});
